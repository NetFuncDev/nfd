program rate_limiter {
	int token = 100;
	entry {
		match_flow {f[tag]==1}
		action_state{
			timer(1, 5000 , 1);
			token =10000;
		}
	}
	entry {
		match_flow {f[tag]==0}
		match_state {token<f[iplen]}
		action_flow{f[dip]=DROP;}
	}
	entry {
		match_flow {f[tag]==0}
		match_state {token>=f[iplen]}
		action_state {token = token- f[iplen]; }
	}
}
